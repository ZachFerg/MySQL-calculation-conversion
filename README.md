            THEN
                'Scanning'
            WHEN
                (PL.date100ProofIn

                    AND SL.moneyInPlantDate01 IS NULL)
            THEN
                'Waiting On Orders'
            WHEN PL.date090ProofShip IS NOT NULL THEN 'Proof Return'

            WHEN PL.date165aProofPrinting IS NOT NULL THEN 'Packing'

            WHEN PL.date020Upload IS NOT NULL THEN 'Data'

            ELSE 'Not In'
